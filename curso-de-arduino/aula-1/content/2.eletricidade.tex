\begin{frame}{Ligação Metálica}
  \begin{columns}
    \column{0.5\textwidth}

\begin{frame}{Tensão Elétrica}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
      \item Quando dois corpos possuem \textbf{cargas elétricas diferentes}, surge uma \textbf{diferença de potencial elétrico}
      \item Essa diferença de potencial é chamada de \textbf{tensão elétrica}
      \item A tensão pode ser interpretada como uma espécie de \textbf{pressão elétrica}
      \item Ela tende a \textbf{impulsionar o movimento de elétrons} entre dois pontos
    \end{itemize}
    \column{0.5\textwidth}
    \begin{figure}
      \centering
      \includegraphics[width=0.8\linewidth]{tension.png}
      \caption{Diferença de carga entre dois corpos metálicos.}
    \end{figure}
  \end{columns}
\end{frame}

\note{
  Imagine duas caixas d'água em alturas diferentes: a água naturalmente desce da mais alta para a mais baixa. No caso da eletricidade, em vez de altura, temos a força elétrica: como os elétrons têm a mesma carga, eles se repelem e acabam sendo empurrados e do outro lado são atraidos. Essa “diferença” que causa o movimento é a tensão.
}

\begin{frame}{Tensão Elétrica}
  \begin{block}{Definição}
    \Large
    Tensão elétrica (V), também conhecida como \textbf{diferença de potencial (DDP)},
    é a diferença de potencial elétrico entre dois pontos.
  \end{block}
  \vspace{0.5em}
  \centering
  \large
  Sua unidade de medida é o \textbf{volt (V)}.
\end{frame}

\begin{frame}{Corrente Elétrica}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
      \item Quando dois pontos possuem \textbf{tensão elétrica}, pode surgir movimento de cargas
      \item Ao conectarmos esses pontos por um \textbf{condutor}, criamos um caminho para os elétrons
      \item O \textbf{movimento ordenado de elétrons} através do condutor é chamado de \textbf{corrente elétrica}
      \item A corrente elétrica permite alimentar dispositivos elétricos e eletrônicos
    \end{itemize}
    \column{0.5\textwidth}
    \begin{figure}
      \centering
      \includegraphics[width=0.8\linewidth]{current.png}
      \caption{Movimento de elétrons (ou, simultaneamente, lacunas) em um condutor.}
    \end{figure}
  \end{columns}
\end{frame}

\note{
  Na analogia das caixas d'água, a corrente é o fluxo de água no cano: quanta água passa por segundo. Nos circuitos, isso corresponde à quantidade de elétrons em movimento.
  \\
  Um detalhe importante: por convenção, dizemos que a corrente vai do positivo para o negativo. Isso pode ser entendido como o movimento de “lacunas” (ausência de elétrons) nesse sentido. Já os elétrons reais se movem no sentido oposto, do negativo para o positivo.
}

\begin{frame}{Corrente Elétrica}
  \begin{block}{Definição}
    \Large
    Corrente elétrica (I) é o fenômeno físico onde os portadores de carga elétrica, como elétrons, são conduzidos pelo interior de algum material em razão da aplicação de uma diferença de potencial elétrico.
  \end{block}
  \vspace{0.5em}
  \centering
  \large
  Sua unidade de medida é o \textbf{ampere (A)}.
\end{frame}

\begin{frame}{Resistência Elétrica}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
      \item Em um circuito, a resistência controla a corrente elétrica
      \item O componente eletrônico responsável por fornecer resistência é o \textbf{resistor}
    \end{itemize}
    \column{0.5\textwidth}
    \begin{figure}
      \centering
      \includegraphics[width=0.6\linewidth]{components/resistors.png}
      \caption{Resistores.}
    \end{figure}
  \end{columns}
  \vspace{0.5em}
  \begin{block}{Definição}
    \large
    Resistência elétrica (R) é a propriedade de um material que \textbf{dificulta a passagem de corrente elétrica} através dele.
  \end{block}
  \vspace{0.5em}
  \centering
  \large
  Sua unidade de medida é o \textbf{ohm ($\Omega$)}.
\end{frame}

\begin{frame}{Lei de Ohm}
  \begin{columns}
    \column{0.6\textwidth}
    \begin{block}{Lei}
      A relação entre \textbf{tensão (V)}, \textbf{corrente (I)} e \textbf{resistência (R)}
      é definida pela \textbf{Lei de Ohm}.
      Ela afirma que a corrente elétrica é \textbf{diretamente proporcional à tensão}
      e \textbf{inversamente proporcional à resistência}.
    \end{block}
    \begin{columns}
      \column{0.33\textwidth}
      \begin{align*}
        V & = R \cdot I
      \end{align*}
      \column{0.33\textwidth}
      \begin{align*}
        I & = \frac{V}{R}
      \end{align*}
      \column{0.33\textwidth}
      \begin{align*}
        R & = \frac{V}{I}
      \end{align*}
    \end{columns}
    \column{0.4\textwidth}
    \begin{figure}
      \centering
      \includegraphics[width=0.8\linewidth]{ohm.jpg}
      \caption{Representação visual da primeira lei de Ohm.}
    \end{figure}
  \end{columns}
\end{frame}

\note{
  Pra lembrar da Lei de Ohm, tem um truque clássico: o triângulo. Coloca o V em cima, e embaixo R e I. Se tampar o que você quer, a fórmula aparece.
  \\
  E tem também a piada clássica: é a lei mais engraçada da física… quem vê, ri.
}

\begin{frame}{Questão 1}
  \textbf{(UNEB-BA)}
  Um resistor ôhmico, quando submetido a uma ddp de \textbf{40 V}, é atravessado
  por uma corrente elétrica de intensidade \textbf{20 A}.
  \vspace{0.5em}
  Quando a corrente que o atravessa for igual a \textbf{4 A}, qual será a
  \textbf{ddp (em volts)} em seus terminais?

  \vspace{1em}

  \visible<2->{
    \begin{block}{Resolução}
      Pela Lei de Ohm:
      $$R = \frac{V}{I} = \frac{40}{20} = 2\,\Omega$$
      Agora:
      $$V = R \cdot I = 2 \times 4 = 8\,V$$
      \textbf{Resposta: 8 V}
    \end{block}
  }
\end{frame}

\begin{frame}{Potência Elétrica}
  \begin{block}{Definição}
    Potência elétrica é a grandeza que determina a \textbf{quantidade de energia
      fornecida ou consumida por unidade de tempo}.
  \end{block}
  \vspace{0.5em}
  {
    \large
    \centering
    Sua unidade de medida é o \textbf{Watt (W)}.
  }
  \begin{center}
    \Large
    $$P = V \cdot I$$
  \end{center}
  \vspace{0.5em}
  \begin{itemize}
    \item Energia pode ser medida em \textbf{Joules (J)}
    \item Se um dispositivo de \textbf{40 W} funciona por \textbf{1 s}, ele consome \textbf{40 J}
    \item Concessionárias de energia usam \textbf{Watt-hora (Wh)} ou \textbf{kWh}
    \item Exemplo: uma lâmpada de \textbf{40 W} ligada por \textbf{1 h} consome \textbf{40 Wh}
  \end{itemize}
\end{frame}

\note{
  Aqui posso fazer a analogia do moinho e da água.

  Pergunta: o que move mais o moinho? pouca água com muita pressão ou muita água com pouca pressão?

  Resposta: tanto faz. O que importa é a potência hidráulica, que depende da vazão e da “força da água”.
  Se uma aumenta e a outra diminui proporcionalmente, a potência continua a mesma.

  Voltando para eletricidade: acontece algo semelhante.
  A potência depende da tensão e da corrente.

  Também explicar a ideia de energia:
  potência é energia por unidade de tempo.

  Se uma lâmpada de 40 W fica ligada por 1 segundo → 40 joules de energia.

  Mas concessionárias não cobram em joules, e sim em watt-hora (Wh),
  que é potência em watts multiplicada pelo tempo em horas.

  Então:
  40 W $\times$ 1 h = 40 Wh.
}

\begin{frame}{Questão 2}
  \textbf{Uma lâmpada de \textbf{40 W} é ligada em uma tomada de \textbf{127 V}.}
  \vspace{0.5em}
  Qual é a \textbf{corrente elétrica} que passa pela lâmpada?
  \vspace{1em}
  \visible<2->{
    \begin{block}{Resolução}
      Usamos a relação da potência elétrica:
      $$P = V \cdot I$$
      Isolando a corrente:
      $$I = \frac{P}{V}$$
      Substituindo os valores:
      $$I = \frac{40}{127} \approx 0{,}315\ \text{A}$$
      \textbf{Resposta:} aproximadamente \textbf{0,32 A}
    \end{block}
  }
\end{frame}
