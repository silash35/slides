\section{Oficina de Circuitos}

\begin{frame}{LED}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
      \item LED significa \textbf{Light Emitting Diode}
      \item Emite luz quando passa corrente elétrica
      \item Conduz corrente em \textbf{apenas um sentido}
      \item Possui dois terminais:
            \begin{itemize}
              \item Ânodo (+)
              \item Cátodo (-)
            \end{itemize}
      \item Normalmente é usado com um \textbf{resistor em série}
    \end{itemize}
    \column{0.5\textwidth}
    \begin{figure}
      \centering
      \includegraphics[width=\linewidth]{components/led.png}
      \caption{LED}
    \end{figure}
  \end{columns}
\end{frame}

\begin{frame}{Dimensionamento do LED}
  \begin{itemize}
    \item Para proteger o LED, calculamos o resistor em série:
  \end{itemize}
  \begin{equation}
    \displaystyle R = \frac{V_{fonte} - V_{LED}}{I_{LED}}
  \end{equation}
  \begin{itemize}
    \item $V_{fonte}$: tensão da fonte
    \item $V_{LED}$: queda de tensão do LED (depende da cor)
    \item $I_{LED}$: corrente desejada (Geralmente 20mA = 0,02A)
  \end{itemize}
\end{frame}

\begin{frame}{Questão 3}
  Deseja-se ligar um LED \textbf{vermelho} em uma fonte de \textbf{9 V}. Qual deve ser o valor do \textbf{resistor em série}?
  \vspace{1em}
  \visible<2->{
    \begin{block}{Resolução}
      Usamos a fórmula do resistor:
      $$R = \frac{V_{fonte} - V_{LED}}{I_{LED}}$$

      Substituindo os valores:
      $$R = \frac{9 - 2}{0{,}02}$$

      $$R = \frac{7}{0{,}02} = 350\ \Omega$$
    \end{block}
  }
\end{frame}

\begin{frame}{Fotoresistor (LDR)}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
      \item LDR significa \textbf{Light Dependent Resistor}
      \item É um \textbf{sensor de luz}
      \item Sua resistência varia com a iluminação
      \item Mais luz $\rightarrow$ menor resistência
      \item Menos luz $\rightarrow$ maior resistência
    \end{itemize}
